TransitJSON / TransitXML

TransitJSON

This is a draft of TransitJSON, a format for representing transit data (e.g. timetables, routes).
TransitJSON is aiming to be as simple as possible so that non-dev people can easily adopt.

{
    "name|名称": "OPERATOR NAME",  // OPTIONAL
    "lines|系統": [  // REQUIRED
        {
            "name|名称": "LINE NAME",  // OPTIONAL
            "href": "LINK TO AN EXTERNAL FILE FOR THIS LINE"

            // OR INLINE REPRESENTATION
            "location|位置": {  // OPTIONAL
                "href": "LINK TO A GEOJSON FILE"
                // OR INLINE GEOJSON
                "type": "LineString",
                "coordinates": [...]
            },

            "routes|路線": [  // REQUIRED
                {
                    "name|名称": "ROUTE NAME",  // OPTIONAL
                    "href": "LINK TO AN EXTERNAL FILE FOR THIS ROUTE"

                    // OR INLINE REPRESENTATION
                    "location|位置": {  // OPTIONAL
                        "href": "LINK TO A GEOJSON FILE"
                        // OR INLINE GEOJSON
                        "type": "LineString",
                        "coordinates": [...]
                    },

                    "directions|方向": [  // REQUIRED
                        {
                            "name|名称",  // OPTIONAL
                            "href": "LINK TO AN EXTERNAL FILE FOR THIS DIRECTION"

                            // OR INLINE REPRESENTATION
                            "services|運行": [  // OPTIONAL (TIMETABLES BY LINE)
                                {
                                    "name|名称": "SERVICE NAME",  // OPTIONAL
                                    "href": "LINK TO AN EXTERNAL FILE FOR THIS SERVICE"

                                    // OR INLINE REPRESENTATION
                                    "period|運行日|期間": "DATE OF SERVICE (ISO 8601 OR ARBITRARY TEXT)",  // OPTIONAL
                                    "stops|停留所|駅|空港|停泊地|etc.": [  // REQUIRED
                                        "a", "b", "c", "d", ...
                                    ],
                                    "times|日時|時刻": [  // REQUIRED (ISO 8601)
                                        ["09:00", "09:05|09:06", "09:10|09:11", "09:15", ...], // arrival|departure
                                        ["10:00", "10:05", "10:10|10:11", "10:15", ...],
                                        ...
                                    ]
                                },
                                {
                                    // SERVICE
                                },
                                ...
                            ]
                        },
                        {
                             // DIRECTION
                        },
                        ...
                    ]
                },
                {
                    // ROUTE
                },
                ...
            ]
        },
        {
            // LINE
        },
        ...
    ],
    "stops|停留所|駅|etc. ": [
        {
            "name": "STOP NAME" // REQUIRED
            "href": "LINK TO AN EXTERNAL FILE FOR THIS STOP"

            // OR INLINE REPRESENTATION
            "location": { // OPTIONAL
                "lat": latitude,
                "lon": longitude
            },
            "stands|のりば|乗り場|etc.": [
                {
                    "name": "STAND(POLE) NAME", // REQUIRED
                    "href": "LINK TO AN EXTERNAL FILE FOR THIS STAND"

                    // OR INLINE REPRESENTATION
                    "location": { // OPTIONAL
                        "lat": latitude,
                        "lon": longitude
                    },

                    "services|運行": [  // OPTIONAL (TIMETABLES BY STOP)
                        {
                            "name|名称": "SERVICE NAME",  // OPTIONAL
                            "href": "LINK TO AN EXTERNAL FILE FOR THIS SERVICE"

                            // OR INLINE REPRESENTATION
                            "period|運行日|期間": "DATE OF SERVICE (ISO 8601 OR ARBITRARY TEXT)",  // OPTIONAL
                            "times|日時|時刻": [  // REQUIRED (ISO 8601)
                                {
                                    "time": "09:00",
                                    "note": "DESTINATION, ANNOTATIONS, etc."
                                },
                                {
                                    "time": "09:05|09:06", // arrival|departure
                                    "note": "DESTINATION, ANNOTATIONS, etc."
                                },
                                ...
                            ]
                        }
                    ]
                },
                {
                    // STAND
                },
                ...
            ]
        },
        {
            // STOP
        },
        ...
    ]
}
